```javascript
var romanToInt = function(s) {
    let a = new HashMap();
    
    a.put('I', 1);
    a.put('V', 5);
    a.put('X', 10);
    a.put('L', 50);
    a.put('C', 100);
    a.put('D', 500);
    a.put('M', 1000);
    
    let res = a.get(s[0]);
    for(let i = 1; i<s.length; i++){
        let cur = a.get(s[i]);
        let pre = a.get(s[i-1]);
        if(pre<cur){
            res -= (pre*2);
        }
        res += cur;
    }
    return res;
};
```

## Letter Combinations of a Phone Number

> [문제](https://leetcode.com/problems/letter-combinations-of-a-phone-number/submissions/)


### 내 답안

```javascript
/**
 * @param {string} digits
 * @return {string[]}
 */


HashMap = function(){  
    this.map = new Array();
};  

HashMap.prototype = {  
    put : function(key, value){  
        this.map[key] = value;
    },  
    get : function(key){  
        return this.map[key];
    },  
};

var letterCombinations = function(digits) {
    let res = new Array();
    let tmp = new Array();
    let a = new HashMap();
    a.put('2', ['a', 'b', 'c']);
    a.put('3', ['d', 'e', 'f']);
    a.put('4', ['g', 'h', 'i']);
    a.put('5', ['j', 'k', 'l']);
    a.put('6', ['m', 'n', 'o']);
    a.put('7', ['p', 'q', 'r', 's']);
    a.put('8', ['t', 'u', 'v']);
    a.put('9', ['w', 'x', 'y', 'z']);
    
    if(digits.length === 0){
        return [];
    }
    
    if(digits.length < 2){
        return a.get(digits);
    }
    
    var combinations = [''];
    
    
    for (var i = 0; i < digits.length; i++) {
        var digit = digits[i];
        var letters = a.get(digit);
        var tempArray = [];
        
        // skip if invalid digit
        if (letters === undefined) continue;
        
        // for as many times as there are letters (eg. 'abc' => 3 times)
        for (var j = 0; j < letters.length; j++) {
            var letterToAdd = letters[j];
            // for as many times as there are existing combinations
            // (eg. ['ad', 'bd', 'cd'] => 3 times)
            for (var k = 0; k < combinations.length; k++) {
                var combination = combinations[k];
                tempArray.push(combination + letterToAdd);
            }
        }
        combinations = tempArray;
    }
    return combinations;
};
```

## Find the Town Judge

> [문제](https://leetcode.com/problems/find-the-town-judge/
)


### 내 답안



```javascript
/**
 * @param {number} N
 * @param {number[][]} trust
 * @return {number}
 */
var findJudge = function(N, trust) {
    const trusted = new Map();
    const all = new Set();
    
    for(let i = 0; i<trust.length; i++){
        if(!trusted.has(trust[i][1])){
            all.add(trust[i][0]);
            trusted.set(trust[i][1], 1)
        }else{
            all.add(trust[i][0]);
            const value = trusted.get(trust[i][1])+1;
            trusted.set(trust[i][1], value)
        }
    }
    
    
    for (let [x, t] of trusted.entries())
        if (!all.has(x) && t == N - 1) 
            return x;
    
    
    return N == 1 ? 1 : -1;
    

};
```
